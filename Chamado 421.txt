select i.s_part_number,
       i.s_descricao,
       n.id_ncm,
       cla.s_descricao_longa,
       i.d_cadastro,
       cla.d_classificacao,
       u.s_nome,
       u.id_usuario,
       i.d_cadastro
  from rgc_item          i,
       rgc_classificacao cla,
       rgc_ncm           n,
       sec_usuario       u,
       rgc_cliente       c
 where i.id_item = cla.id_item
   and n.id_ncm = cla.id_ncm
   and u.id_usuario = cla.id_usuario
   and i.id_cliente = c.id_cliente
   and i.id_item not in (select id_item from rgc_item_catalogo)
   and i.id_cliente in (select id_cliente from rgc_item where S_RAZAO_SOCIAL Like 'Drager%');
